let spinCount = 0;

const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");
const radius = 250;

// gambar roda
function drawWheel() {
  const angle = 2 * Math.PI / data.length;
  data.forEach((item, i) => {
    ctx.beginPath();
    ctx.moveTo(radius, radius);
    ctx.arc(radius, radius, radius, i * angle, (i + 1) * angle);
    ctx.fillStyle = i % 2 === 0 ? "#6ec6ff" : "#81c784";
    ctx.fill();
  });
}
drawWheel();

// fungsi spin
function spin() {
  if (spinCount >= winners.length) {
    alert("Semua pemenang sudah keluar!");
    return;
  }

  const target = winners[spinCount];
  const index = data.indexOf(target);

  if (index === -1) {
    alert("Data tidak cocok!");
    return;
  }

  const anglePerItem = 360 / data.length;
  const targetAngle = 360 - (index * anglePerItem);
  const extra = 360 * (5 + Math.random());

  canvas.style.transition = "transform 5s ease-out";
  canvas.style.transform = `rotate(${extra + targetAngle}deg)`;

  setTimeout(() => {
    alert("Pemenang: " + target);
    spinCount++;
  }, 5000);
}
